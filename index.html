// alumnos ya marcados en esta sesión function openQRScanner() { if (!claseCursoId || !claseMateriaId || !claseFecha) { toast('Seleccioná curso, materia y fecha primero'); return; } if (typeof jsQR === 'undefined') { toast('Librería QR no disponible — verificá conexión'); return; } _qrScannedIds.clear(); document.getElementById('qr-log').innerHTML = ''; document.getElementById('qr-scan-status').textContent = 'Apuntá la cámara al QR del alumno'; document.getElementById('qr-scanner-overlay').style.display = 'flex'; navigator.mediaDevices.getUserMedia({ video: { facingMode: 'environment', width: { ideal: 1280 }, height: { ideal: 720 } } }).then(stream => { _qrStream = stream; const video = document.getElementById('qr-video'); video.srcObject = stream; video.play(); video.addEventListener('loadedmetadata', startQRScan); }).catch(err => { toast('No se pudo acceder a la cámara: ' + err.message); closeQRScanner(); }); } function startQRScan() { const video = document.getElementById('qr-video'); const canvas = document.getElementById('qr-canvas'); const ctx = canvas.getContext('2d'); function tick() { if (video.readyState === video.HAVE_ENOUGH_DATA) { canvas.width = video.videoWidth; canvas.height = video.videoHeight; ctx.drawImage(video, 0, 0, canvas.width, canvas.height); const imageData = ctx.getImageData(0, 0, canvas.width, canvas.height); const code = jsQR(imageData.data, imageData.width, imageData.height, { inversionAttempts: 'dontInvert' }); if (code) processQRCode(code.data); } _qrAnimFrame = requestAnimationFrame(tick); } _qrAnimFrame = requestAnimationFrame(tick); } function processQRCode(data) { // QR format: "PD_ALUMNO_{alumnoId}" if (!data.startsWith('PD_ALUMNO_')) return; const alumnoId = parseInt(data.replace('PD_ALUMNO_', '')); if (!alumnoId || isNaN(alumnoId)) return; // Already scanned this session? if (_qrScannedIds.has(alumnoId)) return; const alumno = state.alumnos.find(a => a.id === alumnoId); if (!alumno) { showQRFeedback('❓ Alumno no encontrado', false); return; } // Mark as P if (!asistTemp[alumnoId]) asistTemp[alumnoId] = { estado: null, nota: null }; asistTemp[alumnoId].estado = 'P'; // Update DOM buttons if visible ['P','A','T'].forEach(e => { const btn = document.getElementById(`pat_${alumnoId}_${e}`); if (btn) btn.className = `pat-btn${e === 'P' ? ' active-P' : ''}`; }); updateAsistCounter(); markUnsaved(); _qrScannedIds.add(alumnoId); // Feedback flash showQRFeedback(alumno.nombre, true); // Add to log const log = document.getElementById('qr-log'); if (log) { const chip = document.createElement('span'); chip.style.cssText = 'background:#27ae60;color:#fff;border-radius:6px;padding:3px 8px;font-size:.75rem;font-weight:600'; chip.textContent = '✓ ' + alumno.nombre.split(' ')[0]; log.appendChild(chip); } // Status document.getElementById('qr-scan-status').textContent = `${_qrScannedIds.size} marcado${_qrScannedIds.size !== 1 ? 's' : ''} — seguí escaneando`; // Vibrate if available if (navigator.vibrate) navigator.vibrate(60); } function showQRFeedback(nombre, success) { const fb = document.getElementById('qr-feedback'); const fn = document.getElementById('qr-feedback-name'); fb.querySelector('div').textContent = success ? '✅' : '❓'; fb.style.background = success ? 'rgba(39,174,96,.35)' : 'rgba(192,57,43,.35)'; fn.textContent = nombre; fb.style.display = 'flex'; setTimeout(() => { fb.style.display = 'none'; }, 1200); } function closeQRScanner() { if (_qrAnimFrame) { cancelAnimationFrame(_qrAnimFrame); _qrAnimFrame = null; } if (_qrStream) { _qrStream.getTracks().forEach(t => t.stop()); _qrStream = null; } document.getElementById('qr-scanner-overlay').style.display = 'none'; // Auto-save if anything was scanned if (_qrScannedIds.size > 0) { guardarClase(); toast(`✅ ${_qrScannedIds.size} alumnos marcados y guardados`); } } /* ============================================================ QR CODE GENERATOR (para perfil del alumno) Genera un QR SVG simple usando una matriz de módulos. Usa una implementación mínima sin dependencias externas. ============================================================ */ function generateQRDataURL(text) { // Use a canvas + jsQR approach in reverse isn't possible. // Instead, encode as a URL to a QR API (works offline via data URI approach). // We'll use the Google Charts QR API as fallback, but primarily build // a simple SVG QR using a minimal QR encoder. // For simplicity and reliability, use an img pointing to quickchart.io // — this requires internet, fine for printing. return `https://quickchart.io/qr?text=${encodeURIComponent(text)}&size=200&margin=2`; } function showAlumnoQR(alumnoId) { const a = state.alumnos.find(x => x.id === alumnoId); if (!a) return; const qrText = `PD_ALUMNO_${alumnoId}`; const qrUrl = generateQRDataURL(qrText); openModal(`QR · ${escHtml(a.nombre)}`, `
